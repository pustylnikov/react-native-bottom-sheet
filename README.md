### Install
using npm
```text
npm install @anvilapp/react-native-bottom-sheet --save
```
or using yarn
```text
yarn add @anvilapp/react-native-bottom-sheet
```

### Usage example
```js
import BottomSheet, {ComposingTypes, PositionTypes} from '@anvilapp/react-native-bottom-sheet';

<BottomSheet
    visible={visible}
    ...
>
    Your content here...
</BottomSheet>
```
